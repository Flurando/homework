  t = linspace(0, 2*pi, 100);

  x_t = sin(3*t).*cos(t);
  y_t = sin(3*t).*sin(t);
  x_line = linspace(min(x_t), max(x_t), 100);
  y_line = 2*x_line - 0.5;

  figure;

  plot(x_t, y_t, 'b-', 'LineWidth', 2);
  hold on;

  plot(x_line, y_line, 'r--', 'LineWidth', 2);
  hold on;
  for i = 1:length(x_line)-1
    curve_y_start = sin(3 * linspace(t(1), t(end), 100)) .* sin(linspace(t(1), t(end), 100));
    y_line_start = 2 * x_line(i) - 0.5; 
    y_line_end = 2 * x_line(i+1) - 0.5; 
    if sign(curve_y_start(1)) ~= sign(y_line_start) || sign(curve_y_start(1)) ~= sign(y_line_end)
      for j = 1:length(t)
        y_line_at_x = 2 * x_t(j) - 0.5;
        if abs(x_t(j) - x_line(i)) < 0.01
          inter_x = x_t(j);
          inter_y = y_t(j);
          if inter_y > y_line_at_x
            plot(inter_x, inter_y, 'go', 'MarkerFaceColor', 'g', 'MarkerSize', 8, 'DisplayName', 'Above Line Intersection');
  	  hold on;
          else
            plot(inter_x, inter_y, 'ro', 'MarkerFaceColor', 'r', 'MarkerSize', 8, 'DisplayName', 'Below Line Intersection');
  	  hold on;
          end
        end
      end
    end
  end
  xlabel('x');
  ylabel('y');
  title('Parametric and Linear Equations');
  legend('y = sin(3t)sin(t)', 'y = 2x + 0.5');
